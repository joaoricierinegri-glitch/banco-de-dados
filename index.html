// Controle 3D, Som e Redirecionamento const scene = document.getElementById('scene'); const card = document.getElementById('card'); const btn = document.getElementById('exploreBtn'); const sound = document.getElementById('clickSound'); document.addEventListener('mousemove', (e) => { const xAxis = (window.innerWidth / 2 - e.pageX) / 20; const yAxis = (window.innerHeight / 2 - e.pageY) / 20; scene.style.transform = `rotateY(${xAxis}deg) rotateX(${yAxis}deg)`; const rect = card.getBoundingClientRect(); card.style.setProperty('--mouse-x', `${e.clientX - rect.left}px`); card.style.setProperty('--mouse-y', `${e.clientY - rect.top}px`); }); btn.addEventListener('click', () => { // Toca o som futurista sound.currentTime = 0; sound.play(); // Feedback visual de carregamento btn.innerText = "🚀 Decolando..."; btn.style.filter = "brightness(1.5)"; // Adiciona um efeito de "sumiço" no card para parecer uma viagem espacial card.style.transition = "all 0.8s ease-in"; card.style.opacity = "0"; card.style.transform = "translateZ(500px) scale(0.5)"; // Redireciona após 1.2 segundos (tempo da animação) setTimeout(() => { window.location.href = "https://www.visitbrasil.com/pt/"; }, 1200); });
